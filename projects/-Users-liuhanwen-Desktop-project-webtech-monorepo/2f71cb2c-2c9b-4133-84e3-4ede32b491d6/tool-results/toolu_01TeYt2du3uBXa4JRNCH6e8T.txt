   232→<script>
   233→import multiselect from '@suadelabs/vue3-multiselect';
   234→import '@suadelabs/vue3-multiselect/dist/vue3-multiselect.esm.css';
   235→import AppCardMessage from '../AppCardMessage/AppCardMessage.vue';
   236→import AppCheckBox from '../AppCheckBox/AppCheckBox.vue';
   237→
   238→export default {
   239→  name: 'AppDropdown',
   240→  components: {
   241→    multiselect,
   242→    AppCheckbox: AppCheckBox,
   243→    AppCardMessage,
   244→  },
   245→  props: {
   246→    trackBy: {
   247→      type: String,
   248→      required: false,
   249→      default: 'value',
   250→    },
   251→    options: {
   252→      type: Array,
   253→      required: true,
   254→    },
   255→    modelValue: {
   256→      type: [String, Object, Boolean, Number, Array],
   257→      required: true,
   258→    },
   259→    disabled: {
   260→      type: Boolean,
   261→      default: false,
   262→    },
   263→    searchable: {
   264→      type: Boolean,
   265→      default: false,
   266→    },
   267→    filterable: {
   268→      type: Boolean,
   269→      default: false,
   270→    },
   271→    clearable: {
   272→      type: Boolean,
   273→      default: false,
   274→    },
   275→    placeholder: {
   276→      type: String,
   277→      default: '',
   278→    },
   279→    openDirection: {
   280→      type: String,
   281→      default: 'auto', // auto, top, bottom
   282→    },
   283→    multiple: {
   284→      type: Boolean,
   285→      default: false,
   286→      required: false,
   287→    },
   288→    groupSelect: {
   289→      type: Boolean,
   290→      default: false,
   291→      required: false,
   292→    },
   293→    selectAllLabel: {
   294→      type: String,
   295→      default: '',
   296→      required: false,
   297→    },
   298→    selectNoneLabel: {
   299→      type: String,
   300→      default: '-',
   301→      required: false,
   302→    },
   303→    disabledLabel: {
   304→      type: String,
   305→      default: '-',
   306→      required: false,
   307→    },
   308→    showLabels: {
   309→      type: Boolean,
   310→      default: false,
   311→    },
   312→    showErrorMessage: {
   313→      type: Boolean,
   314→      default: true,
   315→    },
   316→    autoScroll: {
   317→      type: Boolean,
   318→      default: false,
   319→    },
   320→    maxHeight: {
   321→      type: Number,
   322→      default: 300,
   323→    },
   324→    isError: {
   325→      type: Boolean,
   326→      default: null,
   327→    },
   328→    iconClass: {
   329→      type: String,
   330→      default: '',
   331→      required: false,
   332→    },
   333→    disableSingleSelect: {
   334→      type: Boolean,
   335→      default: true,
   336→    },
   337→    styleType: {
   338→      type: String,
   339→      default: 'gui-style-1',
   340→    },
   341→    customizedLabel: {
   342→      type: Boolean,
   343→      default: false,
   344→    },
   345→    showDisplayingValueWhenDisabled: {
   346→      type: Boolean,
   347→      default: false,
   348→    },
   349→    extraBeforeListOptions: {
   350→      type: Array,
   351→      default: () => [],
   352→    },
   353→    loading: {
   354→      type: Boolean,
   355→      default: false,
   356→    },
   357→    hasMore: {
   358→      type: Boolean,
   359→      default: false,
   360→    },
   361→    loadError: {
   362→      type: Boolean,
   363→      default: false,
   364→    },
   365→  },
   366→  emits: ['update:modelValue', 'toggleDropdown', 'optionClick', 'loadMore'],
   367→  data() {
   368→    return {
   369→      dropdownOptions: [],
   370→      MAX_DISPLAYING_LENGTH: 20,
   371→      loadMoreObserver: null,
   372→    };
   373→  },
   374→  computed: {
   375→    vModel: {
   376→      set(value) {
   377→        if (this.groupSelect) {
   378→          this.$emit('update:modelValue', value);
   379→        } else if (this.multiple && !this.groupSelect) {
   380→          this.$emit(
   381→            'update:modelValue',
   382→            value.map((item) => item.value),
   383→          );
   384→        } else {
   385→          if (value.value instanceof Function) {
   386→            this.$emit('update:modelValue', value.default);
   387→            return;
   388→          }
   389→
   390→          this.$emit('update:modelValue', value.value);
   391→        }
   392→      },
   393→      get() {
   394→        if (this.groupSelect) {
   395→          return this.modelValue;
   396→        }
   397→
   398→        if (this.multiple) {
   399→          return this.options.filter((item) => this.modelValue.includes(item.value));
   400→        }
   401→        const singleSelectValue = this.options.filter((item) => {
   402→          if (item.value instanceof Function) {
   403→            return item.value(this.modelValue);
   404→          }
   405→
   406→          return JSON.stringify(item.value) === JSON.stringify(this.modelValue);
   407→        })[0];
   408→
   409→        return singleSelectValue;
   410→      },
   411→    },
   412→    isSelectingNone() {
   413→      return this.vModel.length === 0;
   414→    },
   415→    isSelectingAll() {
   416→      if (this.groupSelect) {
   417→        return this.vModel.length === this.groupAllItem.length;
   418→      }
   419→
   420→      return this.vModel.length === this.options.length;
   421→    },
   422→    isSelectIndeterminate() {
   423→      if (this.groupSelect) {
   424→        return this.vModel.length !== this.groupAllItem.length && this.vModel.length !== 0;
   425→      }
   426→
   427→      return this.vModel.length !== this.options.length && this.vModel.length !== 0;
   428→    },
   429→    groupValues() {
   430→      return this.groupSelect ? 'groupValues' : '';
   431→    },
   432→    groupLabel() {
   433→      return this.groupSelect ? 'groupLabel' : '';
   434→    },
   435→    groupAllItem() {
   436→      if (!this.groupSelect) {
   437→        return [];
   438→      }
   439→
   440→      const result = [];
   441→      this.options.forEach((element) => {
   442→        result.push(...element.groupValues);
   443→      });
   444→      return result;
   445→    },
   446→    displayingValue() {
   447→      if (this.disabled && !this.showDisplayingValueWhenDisabled) {
   448→        return this.disabledLabel;
   449→      }
   450→
   451→      if (this.isSelectingNone) {
   452→        return this.selectNoneLabel;
   453→      }
   454→
   455→      if (this.isSelectingAll) {
   456→        return this.selectAllLabel;
   457→      }
   458→
   459→      let a = [];
   460→      this.vModel.forEach((element, index) => {
   461→        if (a.join(', ').length <= this.MAX_DISPLAYING_LENGTH) {
   462→          a.push(element.text);
   463→        }
   464→      });
   465→      const remainingAmount = this.vModel.length - a.length;
   466→
   467→      a =
   468→        remainingAmount !== 0
   469→          ? `${a.join(', ').slice(0, this.MAX_DISPLAYING_LENGTH)}...,+${remainingAmount}`
   470→          : a.join(', ');
   471→
   472→      return a;
   473→    },
   474→    vModelValues() {
   475→      return this.vModel.map((item) => item.value);
   476→    },
   477→  },
   478→  watch: {
   479→    options(value) {
   480→      this.dropdownOptions = [...value];
   481→    },
   482→  },
   483→  mounted() {
   484→    this.dropdownOptions = [...this.options];
   485→  },
   486→  beforeUnmount() {
   487→    this.teardownLoadMoreObserver();
   488→  },
   489→  methods: {
   490→    isSelectIndeterminateOfGroup(label) {
   491→      if (this.isSelectAllValuesOfGroup(label)) {
   492→        return false;
   493→      }
   494→      const values = this.options.find((item) => item.groupLabel === label).groupValues.map((item) => item.value);
   495→      return values.some((option) => this.vModelValues.includes(option));
   496→    },
   497→    isSelectAllValuesOfGroup(label) {
   498→      const values = this.options.find((item) => item.groupLabel === label).groupValues.map((item) => item.value);
   499→      return values.every((option) => this.vModelValues.includes(option));
   500→    },
   501→    selectAll() {
   502→      if (this.groupSelect && !this.isSelectingAll) {
   503→        this.vModel = this.groupAllItem;
   504→      } else if (!this.groupSelect && !this.isSelectingAll) {
   505→        this.vModel = this.options;
   506→      } else {
   507→        this.vModel = [];
   508→      }
   509→    },
   510→    scrollToSelected() {
   511→      const target = this.$refs.multiselect.$el.querySelector('span.multiselect__option--selected');
   512→      if (target) {
   513→        target.scrollIntoView();
   514→      }
   515→    },
   516→    toggleDropdown(value) {
   517→      this.$emit('toggleDropdown', value);
   518→
   519→      if (value && !this.multiple && this.autoScroll) {
   520→        this.$nextTick(() => {
   521→          this.scrollToSelected();
   522→        });
   523→      }
   524→
   525→      // Setup/teardown infinite scroll observer
   526→      if (value && this.hasMore) {
   527→        this.$nextTick(() => {
   528→          this.setupLoadMoreObserver();
   529→        });
   530→      } else {
   531→        this.teardownLoadMoreObserver();
   532→      }
   533→    },
   534→    setupLoadMoreObserver(retryCount = 0) {
   535→      if (this.loadMoreObserver) {
   536→        return;
   537→      }
   538→
   539→      const trigger = this.$refs.loadMoreTrigger;
   540→      if (!trigger) {
   541→        if (retryCount < 5) {
   542→          setTimeout(() => {
   543→            this.setupLoadMoreObserver(retryCount + 1);
   544→          }, 50);
   545→        }
   546→        return;
   547→      }
   548→
   549→      this.loadMoreObserver = new IntersectionObserver(
   550→        (entries) => {
   551→          const entry = entries[0];
   552→          if (entry.isIntersecting && this.hasMore && !this.loading) {
   553→            this.$emit('loadMore');
   554→          }
   555→        },
   556→        {
   557→          root: this.$refs.multiselect?.$el?.querySelector('.multiselect__content-wrapper'),
   558→          threshold: 0.1,
   559→        },
   560→      );
   561→
   562→      this.loadMoreObserver.observe(trigger);
   563→    },
   564→    teardownLoadMoreObserver() {
   565→      if (this.loadMoreObserver) {
   566→        this.loadMoreObserver.disconnect();
   567→        this.loadMoreObserver = null;
   568→      }
   569→    },
   570→    selectExtraBeforeListOption(item) {
   571→      item.selected = !item.selected;
   572→    },
   573→    handleOptionClick(option) {
   574→      this.$emit('optionClick', option);
   575→    },
   576→  },
   577→};
